node.js  &vue.js
https://nodejs.org/en/
npm config set -g registry https://registry.npm.taobao.org  #npm 淘宝镜像
npm install -g webpack  #安装webpack
npm install -g vue-cli  #vue-cli 全局安装
vue-init webpack frontend #创建一个叫frontend的前端项目
cd frontend #进入 
npm install	#更新
npm run build	#打包到dist
npm run dev #运行
